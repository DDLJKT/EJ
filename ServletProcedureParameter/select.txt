CREATE PROCEDURE retrieve_detail(IN student VARCHAR(255))
BEGIN
    SET @sql = CONCAT('SELECT id, name FROM ', student);
    PREPARE stmt FROM @sql;
    EXECUTE stmt;
    DEALLOCATE PREPARE stmt;
END//
DELIMITER ;
